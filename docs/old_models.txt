class FinancialProduct(Base):
    __tablename__ = "financial_products"

    id = Column(Integer, primary_key=True)
    account_id = Column(
        Integer,
        ForeignKey("accounts.id", ondelete="CASCADE"),
        nullable=False,
        index=True,
    )
    product_type = Column(
        Enum(AccountType),
        nullable=False,
        index=True,
        comment="CREDIT, DEBIT, INVESTMENT, LOAN, MIXED",
    )
    institution = Column(
        Enum(FinancialInstitutes),
        nullable=False,
        default=FinancialInstitutes.BBVA,
    )
    # En caso de no hacer match agregarlo aqui y marcar el enum como other
    other_institution = Column(String(100), nullable=True)
    card_number = Column(String(4), nullable=True)  # ultimos digitos
    client_number = Column(String(5), nullable=True)  # ultimos digitos
    bank_details = Column(
        EncryptedJSON(settings.encryption_key_v1),
        nullable=False,
        comment="E.g.: {'clabe (ultimos digitos)': '...'}",
    )

    # Relationships con lazy loading optimizado
    statements = relationship(
        "AccountStatement",
        back_populates="product",
        cascade="all, delete-orphan",
        passive_deletes=True,
        lazy="select",  # Cambiado de default para ser explícito
        order_by="AccountStatement.end_date.desc()",  # Orden por defecto más útil
    )

    account = relationship(
        "Account", back_populates="financial_products", lazy="select"
    )

    __table_args__ = (
        Index("ix_product_search", "institution", "product_type"),
        Index("ix_card_last4", "card_number"),
        Index("ix_client_last5", "client_number"),
        Index(
            "ix_account_product_type", "account_id", "product_type"
        ),  # Índice compuesto útil
        # Constraint para asegurar que other_institution se use solo cuando sea necesario
        CheckConstraint(
            "(institution != 'OTHER') OR (institution = 'OTHER' AND other_institution IS NOT NULL)",
            name="check_other_institution_required",
        ),
    )


class AccountStatement(Base):
    __tablename__ = "account_statements"

    id = Column(Integer, primary_key=True)
    account_id = Column(
        Integer,
        ForeignKey("accounts.id"),
        nullable=True,
        index=True,
    )
    product_id = Column(
        Integer,
        ForeignKey("financial_products.id", ondelete="CASCADE"),
        nullable=False,
        index=True,
    )
    file_storage_id = Column(
        Integer,
        ForeignKey("file_storage.id", ondelete="SET NULL"),
        nullable=True,  # Cambiado a nullable para evitar orphans
    )

    start_date = Column(DateTime, nullable=False, index=True)  # Agregado índice
    end_date = Column(DateTime, nullable=False, index=True)

    # AI-processed data
    summary_json = Column(JSON, comment="Normalized structured data")
    """Example:
    ```
    {
        "institution": "BBVA",
        "masked_number": "****1234",
        "account_type": "premium",
        "opening_date": "2023-01-15",
        "credit_limit": 15200.00,  # For credit products
    }
    ```
    """
    processed_at = Column(DateTime, default=datetime.now, index=True)  # Agregado índice
    summary = Column(Text, comment="Summary of the statement", nullable=True)
    # Relationships optimizadas
    transactions = relationship(
        "PurchaseSummary",
        back_populates="account_statement",
        cascade="all, delete-orphan",
        passive_deletes=True,
        lazy="select",
        order_by="PurchaseSummary.date.desc()",  # Orden más útil por fecha
    )

    product = relationship(
        "FinancialProduct", back_populates="statements", lazy="select"
    )

    file_storage = relationship(
        "FileStorage", back_populates="account_statements", lazy="select"
    )

    __table_args__ = (
        Index("ix_statement_date_range", "product_id", "start_date", "end_date"),
        Index("ix_statement_processed", "processed_at", "product_id"),
        # Constraint para asegurar fechas lógicas
        CheckConstraint("start_date <= end_date", name="check_date_range_valid"),
        # Evitar duplicados por producto y periodo
        UniqueConstraint(
            "product_id", "start_date", "end_date", name="uq_product_period"
        ),
    )


class PurchaseSummary(TimestampMixin, Base):
    """
    Modelo para almacenar el resumen de transacciones de un usuario.

    El campo `transaction_details` almacena información adicional en formato JSON encriptado.
    Esta información se divide en dos categorías:

    ## CAMPOS QUE SIEMPRE ESTÁN DISPONIBLES (cuando aplican):
    - merchant: Nombre del comercio/establecimiento
    - location: Ubicación de la transacción (ciudad, país)

    ## CAMPOS QUE PUEDEN O NO ESTAR DISPONIBLES:
    - installment_info: Información de pagos a plazos
    - original_amount: Monto original en moneda extranjera
    - fees: Comisiones aplicadas
    - rewards_points: Puntos de recompensa ganados/usados
    - cashback_amount: Cantidad de cashback
    - promotional_info: Información de promociones aplicadas
    - balance_after: Saldo después de la transacción

    ## INFORMACIÓN DE MESES SIN INTERESES:
    - installment_info.current_installment: Mes actual (ej. 3)
    - installment_info.total_installments: Total de meses (ej. 12)
    - installment_info.remaining_installments: Meses restantes (ej. 9)
    - installment_info.monthly_amount: Pago mensual
    - installment_info.is_interest_free: Si es sin intereses
    - installment_info.start_date: Fecha de inicio del plan
    - installment_info.next_payment_date: Fecha del próximo pago
    """

    __tablename__ = "purchase_summaries"

    id = Column(Integer, primary_key=True)

    # Relaciones con constraints mejorados
    account_id = Column(
        Integer,
        ForeignKey("accounts.id", ondelete="CASCADE"),
        nullable=False,
        index=True,
    )
    statement_id = Column(
        Integer,
        ForeignKey("account_statements.id", ondelete="CASCADE"),
        nullable=False,
        index=True,
    )

    # Campos básicos de la transacción
    purchase = Column(String(500), nullable=False)
    transaction_type = Column(Enum(TransactionType), nullable=False, index=True)
    purchase_type = Column(Enum(PurchaseType), nullable=False, index=True)
    category = Column(Enum(PurchaseCategory), nullable=False, index=True)
    date = Column(DateTime, nullable=False, index=True)
    total_amount = Column(Float, nullable=False)

    # Información adicional de la transacción (JSON encriptado)
    transaction_details = Column(
        EncryptedJSON(settings.encryption_key_v1),
        nullable=True,
        comment="Estructura JSON con información detallada de la transacción",
    )

    currency_type = Column(String(3), nullable=False, default="MXN")
    exchange_rate = Column(Float, nullable=True)

    # Campo computado para análisis en MXN
    amount_mxn = Column(
        Numeric(12, 2),
        Computed(
            "CASE WHEN currency_type = 'MXN' THEN total_amount ELSE total_amount * COALESCE(exchange_rate, 1) END",
            persisted=True,
        ),
        comment="Amount converted to MXN for analysis",
    )

    # Installed fields for better performance

    @property
    def has_installment(self) -> bool:
        return bool(
            self.transaction_details
            and self.transaction_details.get("installment_info")
        )

    @property
    def amount(self) -> Decimal:
        """Monto en MXN. Usa monthly_amount si está disponible"""
        # print(self.transaction_details)
        if self.transaction_details and "installment_info" in self.transaction_details:
            try:
                return Decimal(
                    str(
                        self.transaction_details["installment_info"].get(
                            "monthly_amount", "0.0"
                        )
                    )
                )
            except (TypeError, ValueError, InvalidOperation):
                pass
        return self.amount_mxn

    # Relaciones optimizadas
    account = relationship(
        "Account", back_populates="purchase_summaries", lazy="select"
    )
    account_statement = relationship(
        "AccountStatement",
        back_populates="transactions",
        lazy="select",
    )

    def __repr__(self):
        return f"<PurchaseSummary(id={self.id}, purchase={self.purchase[:50]}..., amount={self.total_amount})>"

    def get_installment_progress(self):
        """
        Retorna información del progreso de pagos a meses sin intereses.

        Returns:
            dict: Información del progreso o None si no aplica
            {
                "current_month": 3,
                "total_months": 12,
                "remaining_months": 9,
                "progress_percentage": 25.0,
                "monthly_payment": 500.00,
                "is_interest_free": True,
                "next_payment_date": "2024-04-15"
            }
        """
        if (
            not self.transaction_details
            or "installment_info" not in self.transaction_details
        ):
            return None

        installment = self.transaction_details["installment_info"]

        if not all(
            key in installment for key in ["current_installment", "total_installments"]
        ):
            return None

        current = installment["current_installment"]
        total = installment["total_installments"]

        return {
            "current_month": current,
            "total_months": total,
            "remaining_months": installment.get(
                "remaining_installments", total - current
            ),
            "progress_percentage": round((current / total) * 100, 2),
            "monthly_payment": installment.get("monthly_amount"),
            "is_interest_free": installment.get("is_interest_free", False),
            "next_payment_date": installment.get("next_payment_date"),
            "plan_type": installment.get("plan_type", "unknown"),
        }

    def get_fee_breakdown(self):
        """
        Retorna el desglose de comisiones y fees.

        Returns:
            dict: Desglose de fees o None si no hay información
        """
        if not self.transaction_details or "fees" not in self.transaction_details:
            return None

        fees = self.transaction_details["fees"]
        total_fees = sum(fees.values()) if isinstance(fees, dict) else 0

        return {**fees, "total_fees": total_fees}

    def is_promotional_transaction(self):
        """
        Verifica si la transacción tiene información promocional.

        Returns:
            bool: True si hay información promocional
        """
        return (
            self.transaction_details
            and "promotional_info" in self.transaction_details
            and self.transaction_details["promotional_info"]
        )

    __table_args__ = (
        # Índices optimizados para consultas comunes
        Index("idx_main_search", "date", "category", "purchase_type"),
        Index("idx_account_date", "account_id", "date"),
        Index("idx_statement_category", "statement_id", "category"),
        Index("idx_amount_analysis", "currency_type", "total_amount"),

        CheckConstraint(

            name="check_exchange_rate_required",
        ),
        CheckConstraint("exchange_rate > 0", name="check_exchange_rate_positive"),
    )

        cascade="all, delete-orphan",
        primaryjoin="FileStorage.id == InsightsModel.file_storage_id",

    def __repr__(self):
